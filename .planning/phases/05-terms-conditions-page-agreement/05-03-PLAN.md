    ```tsx
    import NavBar from "@/components/NavBar"
    import Footer from "@/components/Footer"
    import { termsClauses, lastUpdated } from "@/lib/terms"

    export default function TermsPage() {
      return (
        <>
          <NavBar />
          <main className="mx-auto max-w-3xl px-4 py-16 sm:py-24">
            <div className="rounded-xl bg-card p-6 sm:p-8 shadow-sm ring-1 ring-border">
              {/* Page heading */}
              <h1 className="font-heading text-2xl sm:text-3xl font-bold text-foreground mb-2">
                Terms &amp; Conditions
              </h1>
              <p className="text-xs sm:text-sm text-muted-foreground mb-8">
                Last updated: {lastUpdated}
              </p>

              {/* Clauses */}
              {termsClauses.map((clause, index) => (
                <section key={clause.id}>
                  <h2 className="font-heading text-base sm:text-lg font-semibold mb-2">
                    {clause.title}
                  </h2>
                  <p className="font-sans text-sm sm:text-base leading-relaxed text-muted-foreground">
                    {clause.content}
                  </p>
                  {/* Separator between clauses (skip after last) */}
                  {index < termsClauses.length - 1 && (
                    <hr className="my-6 border-border" />
                  )}
                </section>
              ))}

              {/* Back link */}
              <div className="mt-10">
                <a
                  href="/#form"
                  className="inline-flex items-center gap-1.5 text-primary font-medium hover:underline"
                >
                  &larr; Back to Registration
                </a>
              </div>
            </div>
          </main>
          <Footer />
        </>
      )
    }
    ```

    **Styling details:**
    - `<h1>` page heading: `font-heading text-2xl sm:text-3xl font-bold text-foreground` — Bodoni Moda bold heading per UI-SPEC (D-05 line 60)
    - "Last updated" line: `text-xs sm:text-sm text-muted-foreground` per UI-SPEC
    - Clause `<h2>`: `font-heading text-base sm:text-lg font-semibold` per UI-SPEC — Bodoni Moda semibold subheading
    - Clause `<p>` body: `font-sans text-sm sm:text-base leading-relaxed text-muted-foreground` — Jost body per UI-SPEC (not Playfair Display SC or Karla which are aspirational references — per Pitfall 1)
    - Card: `rounded-xl bg-card p-6 sm:p-8 shadow-sm ring-1 ring-border` — identical to RegistrationForm card styling (UI-SPEC line 172)
    - Separator: `<hr className="my-6 border-border" />` — using native `<hr>` instead of shadcn `<Separator>` to keep this a pure server component with no client bundle. The visual is identical.
    - "Back to Registration" link: `text-primary font-medium hover:underline`, navigates to `/#form` per D-03

    **Key layout:**
    - `<main>` wrapper: `mx-auto max-w-3xl px-4 py-16 sm:py-24` — centered, matches main page content width per D-11
    - Card padding ensures content doesn't touch edges on mobile
    - No `<Separator>` import needed — using native `<hr>` with `border-border` class keeps the server component import-free of client components

    **What to NOT do:**
    - Do NOT add `"use client"` — this is a server component (no interactivity, zero JS shipped)
    - Do NOT hardcode clause text in the page file — import from `lib/terms.ts` per D-05
    - Do NOT use Playfair Display SC or Karla font classes — use `font-heading` and `font-sans` per Pitfall 1
    - Do NOT use shadcn Separator component — native `<hr>` avoids pulling in a client component (ponytail: simpler, avoids client bundle)

        expect(screen.getByTestId("footer")).toBeInTheDocument()
