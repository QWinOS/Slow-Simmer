    export async function POST(request: Request) {

        const { amount, notes } = (await request.json()) as {

        return NextResponse.json({ error: message }, { status: 500 })

    This route is called by the PaymentSection component after the RazorPay checkout modal fires its success handler. It verifies the payment signature server-side, providing immediate UI feedback before the webhook arrives.

    ```typescript
    import { verifyPaymentSignature } from "@/lib/razorpay"
    import { NextResponse } from "next/server"

    export const dynamic = "force-dynamic"

    export async function POST(request: Request) {
      try {
        const {
          razorpay_order_id,
          razorpay_payment_id,
          razorpay_signature,
        } = (await request.json()) as {
          razorpay_order_id: string
          razorpay_payment_id: string
          razorpay_signature: string
        }

        if (!razorpay_order_id || !razorpay_payment_id || !razorpay_signature) {
          return NextResponse.json(
            { error: "Missing payment verification fields" },
            { status: 400 },
          )
        }

        const isValid = verifyPaymentSignature({
          orderId: razorpay_order_id,
          paymentId: razorpay_payment_id,
          signature: razorpay_signature,
        })

        if (!isValid) {
          return NextResponse.json(
            { error: "Invalid payment signature" },
            { status: 400 },
          )
        }

        return NextResponse.json({
          verified: true,
          paymentId: razorpay_payment_id,
        })
      } catch (err) {
        const message = err instanceof Error ? err.message : "Unknown error"
        return NextResponse.json({ error: message }, { status: 500 })
      }
    }
    ```

    Per RESEARCH.md: This client-side verification is for UI feedback only (Pattern 3 comment: "do NOT skip it just because the webhook also verifies"). The webhook is the source of truth for Sheets write and email send.

    Follow the established error handling pattern from locations/route.ts.

  <action>
    Create `app/api/webhooks/razorpay/route.ts` per RESEARCH.md "Complete RazorPay Webhook Handler" (lines 909-1038) and PATTERNS.md "webhooks/razorpay/route.ts" analog (lines 146-182).

    This is the sole source of truth for payment confirmation per D-02. It:
    1. Reads raw body via request.text() — CRITICAL for signature verification
    2. Verifies HMAC-SHA256 signature using RAZORPAY_WEBHOOK_SECRET
    3. Processes payment.captured events only
    4. Checks idempotency via Payment ID (D-06)
    5. Appends registration data to Google Sheets (SHEET-01)
    6. Sends Brevo confirmation email (NOTF-01)
    7. Implements in-memory retry queue for failed Sheets writes (D-07)
    8. Always returns 200 to prevent RazorPay retries (Pitfall 5)

    ```typescript
    import { verifyWebhookSignature } from "@/lib/razorpay"
    import { appendRegistrationRow, checkPaymentIdExists } from "@/lib/sheets-write"
    import { sendConfirmationEmail } from "@/lib/brevo"
    import { NextResponse } from "next/server"

    export const dynamic = "force-dynamic"

    // In-memory retry queue per D-07: maps Payment ID → pending data
    const pendingWrites = new Map<string, {
      row: Parameters<typeof appendRegistrationRow>[0]
      emailParams: Parameters<typeof sendConfirmationEmail>[0]
      retries: number
    }>()

    async function processPendingWrites() {
      for (const [paymentId, pending] of pendingWrites) {
        try {
          await appendRegistrationRow(pending.row)
          pendingWrites.delete(paymentId)
        } catch {
          pending.retries++
          if (pending.retries > 5) {
            console.error(`Giving up on payment ${paymentId} after 5 retries`)
            pendingWrites.delete(paymentId)
          }
        }
      }
    }

    export async function POST(request: Request) {
      // Process queued writes from previous failures (D-07)
      await processPendingWrites()

      try {
        // 1. Read RAW body — BEFORE any other body access (Pitfall 1)
        const rawBody = await request.text()
        const signature = request.headers.get("x-razorpay-signature")

        if (!signature) {
          return NextResponse.json({ error: "Missing signature header" }, { status: 400 })
        }

        const webhookSecret = process.env.RAZORPAY_WEBHOOK_SECRET
        if (!webhookSecret) {
          return NextResponse.json({ error: "Webhook secret not configured" }, { status: 500 })
        }

        // 2. Verify HMAC-SHA256 signature using webhook secret (NOT key_secret — Pitfall 2)
        const isValid = verifyWebhookSignature({
          body: rawBody,
          signature,
          secret: webhookSecret,
        })

        if (!isValid) {
          return NextResponse.json({ error: "Signature mismatch" }, { status: 401 })
        }

        // 3. Parse verified payload
        const event = JSON.parse(rawBody)

        // Only process payment.captured events
        if (event.event !== "payment.captured") {
          return NextResponse.json({ received: true })
        }

        const payment = event.payload.payment.entity
        const paymentId = payment.id
        const notes = payment.notes || {}

        // 4. Idempotency check (D-06, SHEET-03)
        const isDuplicate = await checkPaymentIdExists(paymentId).catch(() => false)
        if (isDuplicate) {
          return NextResponse.json({ received: true, duplicate: true })
        }

        // 5. Prepare registration row data from order notes
        const registrationRow = {
          location: notes.location || "",
          eventDate: notes.eventDate || "",
          eventTime: notes.eventTime || "",
          name: notes.name || "",
          contact: notes.contact || "",
          email: notes.email || "",
          aadhar: notes.aadhar || "",
          bringingGuest: notes.bringingGuest === "true" ? "Yes" : "No",
          guestName: notes.guestName || "",
          guestAge: notes.guestAge || "",
          about: notes.about || "",
          social: notes.social || "",
          paymentStatus: "captured",
          paymentId,
          timestamp: new Date().toISOString(),
        }

        const emailParams = {
          email: notes.email,
          name: notes.name,
          location: notes.location,
          eventDate: notes.eventDate,
          eventTime: notes.eventTime,
        }

        // 6. Write to Sheets — if it fails, queue for retry (D-07)
        try {
          await appendRegistrationRow(registrationRow)
        } catch (sheetsErr) {
          console.error("Sheets append failed, queuing for retry:", sheetsErr)
          pendingWrites.set(paymentId, { row: registrationRow, emailParams, retries: 0 })
        }

        // 7. Send confirmation email (fire-and-forget — don't block on email failure)
        sendConfirmationEmail(emailParams).catch((err) => {
          console.error("Confirmation email failed:", err)
        })

        // 8. Always return 200 to acknowledge receipt (Pitfall 5)
        return NextResponse.json({ received: true })
      } catch (err) {
        const message = err instanceof Error ? err.message : "Unknown error"
        console.error("Webhook error:", message)
        // Always return 200 — RazorPay retries on non-2xx (Pitfall 5)
        return NextResponse.json({ received: true, error: message })
      }
    }
    ```

    CRITICAL IMPLEMENTATION NOTES per RESEARCH.md Pitfalls:

    **Pitfall 1 — Raw body:** The `await request.text()` call MUST come before any attempt to parse the body. If `request.json()` is called first, the body stream is consumed and `.text()` returns an empty string, causing signature verification to fail. The raw body text is passed directly to `verifyWebhookSignature` — never re-stringify it.

    **Pitfall 2 — Webhook secret:** This route uses `RAZORPAY_WEBHOOK_SECRET` (configured in RazorPay Dashboard webhook settings), NOT `RAZORPAY_KEY_SECRET` (API secret). They are different values.

    **Pitfall 3 — Timing-safe comparison:** The `verifyWebhookSignature` function from lib/razorpay.ts uses `crypto.timingSafeEqual()` for constant-time comparison.

    **Pitfall 5 — Always return 200:** The outer catch block returns `{ received: true, error: message }` with status 200, NOT 500. This prevents RazorPay from retrying the webhook for transient errors. The in-memory retry queue (D-07) handles Sheets failures internally.

    **Notes handling:** The registration data is passed as `notes` during order creation (in PaymentSection). RazorPay forwards these notes to the webhook payload as `payment.notes`. All fields are flat strings — no nested objects (per Assumption A1). Field values default to empty string if missing.

    **Type safety:** Use inline type annotations for the pendingWrites Map values. The `Parameters<>` utility type extracts the row/params types from the imported functions, ensuring type coherence if the lib interfaces change.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client browser → POST /api/orders/create | Untrusted client sends amount and notes for order creation |
| Client browser → POST /api/orders/verify | Untrusted client sends payment_id, order_id, and signature |
| RazorPay server → POST /api/webhooks/razorpay | External webhook event enters server — must verify authenticity |
| Webhook handler → Google Sheets API | Server-side authenticated call with write-scope token |
| Webhook handler → Brevo REST API | Server-side authenticated call with API key |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation Plan |
|-----------|----------|-----------|----------|-------------|-----------------|
| T-03-06 | Spoofing | Webhook handler receives forged payment.captured event | critical | mitigate | HMAC-SHA256 signature verification via crypto.timingSafeEqual using RAZORPAY_WEBHOOK_SECRET. Signature checked BEFORE any data processing. Invalid sig returns 401. |
| T-03-07 | Repudiation | Duplicate webhook delivery adds duplicate rows | high | mitigate | Idempotency check (checkPaymentIdExists) BEFORE Sheets append. Duplicate Payment IDs return early with { duplicate: true }. |
| T-03-08 | Tampering | Client calls /orders/verify with forged payment data | medium | mitigate | Server-side HMAC verification uses RAZORPAY_KEY_SECRET — client cannot forge valid signature. Webhook is the source of truth (D-02); /verify is only for UI feedback. |
| T-03-09 | Denial of Service | Webhook endpoint overload from RazorPay retries | low | mitigate | Handler returns 200 even on transient errors (Pitfall 5), preventing RazorPay's exponential retry mechanism. Signature verification is fast (no async ops before check). |
| T-03-10 | Information Disclosure | Aadhar data in order notes visible in RazorPay logs | medium | accept | Aadhar is included in the 15-column Sheets schema per D-05. RazorPay order notes are visible in the RazorPay dashboard but are considered acceptable for MVP scale. All data transmitted over HTTPS. |
